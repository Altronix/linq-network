add_library(mock-libusb mock_libusb.c mock_libusb.h)
target_link_libraries(mock-libusb cmocka stdc++ linq-utils-static)
target_include_directories(mock-libusb PUBLIC .)

function (append_mock_libusb_linker_flags dst)
  string(CONCAT wrap_flags "-Wl,")
  string(CONCAT wrap_flags ${wrap_flags} "--wrap=libusb_init,")
  string(CONCAT wrap_flags ${wrap_flags} "--wrap=libusb_exit,")
  string(CONCAT wrap_flags ${wrap_flags} "--wrap=libusb_open,")
  string(CONCAT wrap_flags ${wrap_flags} "--wrap=libusb_close,")
  string(CONCAT wrap_flags ${wrap_flags} "--wrap=libusb_get_device_list,")
  string(CONCAT wrap_flags ${wrap_flags} "--wrap=libusb_get_string_descriptor_ascii,")
  string(CONCAT wrap_flags ${wrap_flags} "--wrap=libusb_free_device_list,")
  string(CONCAT wrap_flags ${wrap_flags} "--wrap=libusb_get_device_descriptor")
  set(${dst} ${wrap_flags} PARENT_SCOPE)
endfunction()
append_mock_libusb_linker_flags(wrap_flags)

add_executable(test-mock-libusb main.c)
set_target_properties(test-mock-libusb PROPERTIES LINK_FLAGS ${wrap_flags})
target_link_libraries(test-mock-libusb mock-libusb cmocka)
add_test(NAME test-mock-libusb COMMAND test-mock-libusb)
install(TARGETS test-mock-libusb DESTINATION bin)
